program complex_nested;
begin
    total := 0;
    i := 1;
    while i <= 10 do begin
        if (i mod 2) = 0 then begin
            if not (i = 6) then begin
                total := total + i;
                if total > 20 then break
            end
        end;
        i := i + 1
    end;
    if (total > 15) and (i < 12) then
        output := total * 2
    else
        output := total
end.
